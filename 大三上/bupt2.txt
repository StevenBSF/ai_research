					type:"get",
					url:$.url("a/common/student/resumeinfo/getResumeList"),
					data:dataSendJob,
					dataType:"json",

					success:function(res){
						if(res.state==1){
							if(!res.object){
								$.jBox.tip('请先添加简历之后再进行投递！','error',{persistent:true,opacity:0,timeout:1500});

								},1000)

							$('.joinBtn').removeAttr('disabled')
							setTimeout(function(){
								window.location.href=window._config.domain+window._config.jobUrl.url
